26CHARSET = "qpzry9x8gf2tvdw0s3jn54khce6mua7l"

29def bech32_polymod(values: List[int]): 

30 """Internal function that computes the Bech32 checksum.""" 

31 generator = [0x3b6a57b2, 0x26508e6d, 0x1ea119fa, 0x3d4233dd, 0x2a1462b3] 

32 chk = 1 

33 for value in values: 

34 top = chk >> 25 

35 chk = (chk & 0x1ffffff) << 5 ^ value 

36 for i in range(5): 

37 chk ^= generator[i] if ((top >> i) & 1) else 0 

38 return chk

41def bech32_hrp_expand(hrp: str) -> List[int]: 

42 """Expand the HRP into values for checksum computation.""" 

43 return [ord(x) >> 5 for x in hrp] + [0] + [ord(x) & 31 for x in hrp]

46def bech32_verify_checksum(hrp: str, data: List[int]): 

47 """Verify a checksum given HRP and converted data characters.""" 

48 return bech32_polymod(bech32_hrp_expand(hrp) + data) == 1

51def bech32_create_checksum(hrp: str, data: List[int]): 

52 """Compute the checksum values given HRP and data.""" 

53 values = bech32_hrp_expand(hrp) + data 

54 polymod = bech32_polymod(values + [0, 0, 0, 0, 0, 0]) ^ 1 

55 return [(polymod >> 5 * (5 - i)) & 31 for i in range(6)]

58def bech32_encode(hrp: str, data: List[int]): 

59 """Compute a Bech32 string given HRP and data values.""" 

60 combined = data + bech32_create_checksum(hrp, data) 

61 return hrp + '1' + ''.join([CHARSET[d] for d in combined])

64def bech32_decode(bech: str): 

65 """Validate a Bech32 string, and determine HRP and data.""" 

66 if ((any(ord(x) < 33 or ord(x) > 126 for x in bech)) or (bech.lower() != bech and bech.upper() != bech)): 

67 return (None, None) 

68 bech = bech.lower() 

69 pos = bech.rfind('1') 

70 if pos < 1 or pos + 7 > len(bech) or len(bech) > 90: 

71 return (None, None) 

72 if not all(x in CHARSET for x in bech[pos + 1:]): 

73 return (None, None) 

74 hrp = bech[:pos] 

75 data = [CHARSET.find(x) for x in bech[pos + 1:]] 

76 if not bech32_verify_checksum(hrp, data): 

77 return (None, None) 

78 return (hrp, data[:-6])

81def convertbits(data: Union[List[int], bytes], frombits: int, tobits: int, pad: bool = True) -> Union[List[int], None]: 

82 """General power-of-2 base conversion.""" 

83 acc: int = 0 

84 bits: int = 0 

85 ret: List[int] = [] 

86 maxv: int = (1 << tobits) - 1 

87 max_acc = (1 << (frombits + tobits - 1)) - 1 

88 for value in data: 

89 if value < 0 or (value >> frombits): 

90 return None 

91 acc = ((acc << frombits) | value) & max_acc 

92 bits += frombits 

93 while bits >= tobits: 

94 bits -= tobits 

95 ret.append((acc >> bits) & maxv) 

96 if pad: 

97 if bits: 

98 ret.append((acc << (tobits - bits)) & maxv) 

99 elif bits >= frombits or ((acc << (tobits - bits)) & maxv): 

100 return None 

101 return ret

104def decode(hrp: str, addr: str): 

105 """Decode a segwit address.""" 

106 hrpgot, data = bech32_decode(addr) 

107 if hrpgot is None or data is None: 

108 return (None, None) 

109 if hrpgot != hrp: 

110 return (None, None) 

111 decoded = convertbits(data[1:], 5, 8, False) 

112 if decoded is None or len(decoded) < 2 or len(decoded) > 40: 

113 return (None, None) 

114 if data[0] > 16: 

115 return (None, None) 

116 if data[0] == 0 and len(decoded) != 20 and len(decoded) != 32: 

117 return (None, None) 

118 return (data[0], decoded)

121def encode(hrp: str, witver: int, witprog: List[int]): 

122 """Encode a segwit address.""" 

123 converted = convertbits(witprog, 8, 5) 

124 if converted is None: 

125 return None 

126 ret = bech32_encode(hrp, [witver] + converted) 

127 if decode(hrp, ret) == (None, None): 

128 return None 

129 return ret
