147 def compute_contract_address(self, deployer: IAddress, deployment_nonce: int) -> Address: 

148 """Computes the contract address based on the deployer's address and deployment nonce. 

149 

150 Args: 

151 deployer (IAddress): The address of the deployer\n 

152 deployment_nonce (int): The nonce of the deployment 

153 

154 Returns: 

155 Address: The computed contract address as below: 

156 

157 8 bytes of zero + 2 bytes for VM type + 20 bytes of hash(owner) + 2 bytes of shard(owner)""" 

158 deployer_pubkey = deployer.get_public_key() 

159 nonce_bytes = deployment_nonce.to_bytes(8, byteorder="little") 

160 bytes_to_hash = deployer_pubkey + nonce_bytes 

161 contract_pubkey = keccak.new(digest_bits=256).update(bytes_to_hash).digest() 

162 contract_pubkey = bytes([0] * 8) + bytes([5, 0]) + contract_pubkey[10:30] + deployer_pubkey[30:] 

163 return Address(contract_pubkey, deployer.get_hrp())

181def _decode_bech32(value: str) -> Tuple[str, bytes]: 

182 hrp, value_bytes = bech32.bech32_decode(value) 

183 if hrp is None or value_bytes is None: 

184 raise ErrBadAddress(value) 

185 

186 decoded_bytes = bech32.convertbits(value_bytes, 5, 8, False) 

187 if decoded_bytes is None: 

188 raise ErrBadAddress(value) 

189 

190 return hrp, bytes(bytearray(decoded_bytes))

193def get_shard_of_pubkey(pubkey: bytes, number_of_shards: int) -> int: 

194 mask_high = int("11", 2) 

195 mask_low = int("01", 2) 

196 

197 last_byte_of_pubkey = pubkey[31]

202 shard = last_byte_of_pubkey & mask_high 

203 if shard > number_of_shards - 1:
