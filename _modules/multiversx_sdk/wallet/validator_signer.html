    def sign(self, data: bytes) -> bytes:
        try:
            return self._try_sign(data)
        except Exception as err:
            raise CannotSignError() from err



    def _try_sign(self, data: bytes) -> bytes:
        signature = self.secret_key.sign(data)
        return signature
