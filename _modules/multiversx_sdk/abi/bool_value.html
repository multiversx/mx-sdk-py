    def decode_top_level(self, data: bytes):
        if len(data) == 0:
            self.value = False
            return

        if len(data) == 1:
            self.value = self._byte_to_bool(data[0])
            return

        raise ValueError(f"unexpected boolean value: {data}")



    def _byte_to_bool(self, data: int) -> bool:
        if data == TRUE_AS_BYTE:
            return True

        if data == FALS_AS_BYTE:
            return False

        raise ValueError(f"unexpected boolean value: {data}")
